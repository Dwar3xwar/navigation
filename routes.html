Routes

Tidying Up Urls

var stateNavigator = new Navigation.StateNavigator([
  {key: 'inbox', route: ''},
  {key: 'mail', route: 'open'},
  {key: 'compose'}
]);

var stateNavigator = new Navigation.StateNavigator([
  {key: 'inbox', route: ''},
  {key: 'mail', route: 'open', defaultTypes: {id: 'number'}},
  {key: 'compose'}
]);

var stateNavigator = new Navigation.StateNavigator([
  {key: 'inbox', route: '', trackTypes: false},
  {key: 'mail'},
  {key: 'compose'}
]);

Setting Route Parameters

var stateNavigator = new Navigation.StateNavigator([
  {key: 'inbox', route: ''},
  {key: 'mail', route: 'open/{id}', defaultTypes: {id: 'number'}},
  {key: 'compose'}
]);

var stateNavigator = new Navigation.StateNavigator([
  {key: 'inbox', route: ''},
  {key: 'mail'},
  {key: 'compose', route: 'new/{*contacts}', defaultTypes: {contacts: 'numberarray'}}
]);
