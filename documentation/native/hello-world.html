Hello World

The Navigation router uses the underlying native APIs to provide faithful navigation on Android and iOS. It's the only router for React Native that offers native navigation where the stack can be interrogated in JavaScript. The best of both worlds. We're going to build an example with two scenes. The first scene shows a button that says 'Hello'. Clicking the button pushes a second scene onto the stack. This scene displays a button that says 'World'. Clicking it pops the second scene off the stack and returns you to the first. 

var stateNavigator = new StateNavigator([
  {key: 'hello', title: 'Hello'},
  {key: 'world', title: 'World', trackCrumbTrail: true}
]);
stateNavigator.navigate('hello');

Push

import {NavigationContext} from 'navigation-react';

var {hello} = stateNavigator.states;

hello.renderScene = () => (
  <NavigationContext.Consumer>
    {({stateNavigator}) => (
      <TouchableHighlight
        onPress={() => {
          stateNavigator.navigate('world', {size: 20});
        }}>
        <Text>Hello</Text>
      </TouchableHighlight>
    )}
  </NavigationContext.Consumer>
);
var {world} = stateNavigator.states;

world.renderScene = ({size}) => (
  <Text style={{fontSize: data.size}}>World</Text>
);

Pop

world.renderScene = ({size}) => (
  <NavigationContext.Consumer>
    {({stateNavigator}) => (
      <TouchableHighlight        
        onPress={() => {
          stateNavigator.navigateBack(1);
        }}>
        <Text style={{fontSize: size}}>World</Text>
      </TouchableHighlight>
    )}
  </NavigationContext.Consumer>
);
