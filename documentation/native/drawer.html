Drawer

You can use any third-party Drawer component with the Navigation router. 

import DrawerLayout from 'react-native-gesture-handler/DrawerLayout';

<DrawerLayout
  ref={this.ref}
  onDrawerClose={() => {
    if (this.state.selected)
      stateNavigator.navigate('sent');
  }}
  renderNavigationView={() => (
    <View>
      <TouchableHighlight
        onPress={() => {
          this.setState({selected: true});
          this.ref.current.closeDrawer();
      }}>
        <Text>Sent</Text>
      </TouchableHighlight>
    </View>
  )}> 
var stateNavigator = new StateNavigator([
  {key: 'inbox'},
  {key: 'mail', trackCrumbTrail: true},
  {key: 'compose', trackCrumbTrail: true},
  {key: 'sent'}
]);

var {sent} = stateNavigator.states;
sent.renderScene = () => <Sent />;
<NavigationHandler stateNavigator={stateNavigator}>
  <NavigationStack unmountStyle={(from, state) => state === sent ? '' : null} />
</NavigationHandler>
