Priming the Stack

var stateNavigator = new StateNavigator([
  {key: 'inbox', title: 'Inbox'},
  {key: 'mail', title: 'Mail', trackCrumbTrail: true},
  {key: 'attachment', title: 'Attachment', trackCrumbTrail: true}
]);
onPress={() => {
  var url = stateNavigator.fluent(true)
    .navigate('mail', {id: 12})
    .navigate('attachment', {id: 5}).url;
  stateNavigator.navigateLink(url);
}
