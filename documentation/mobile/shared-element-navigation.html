Animating

var MailShared = function(props) {
  return (
    <SharedElementMotion
      {...props}
      onAnimating={(name, ref) => {ref.style.opacity = 0}}
      onAnimated={(name, ref) => {ref.style.opacity = 1}}>
      {(style, name, data) => (
        <div
          key={name}
          style={{
            fontSize: `${size}px`,
            left: style.left,
            top: style.top,
            width: style.width,
            height: style.height,
            position: 'absolute',
          }}>
          {data.title}
        </div>
      )}
    </SharedElementMotion>
  );
}
<NavigationMotion
  unmountedStyle={{opacity: 0}}
  mountedStyle={{opacity: 1}}
  crumbStyle={{opacity: 0}}
  sharedElementMotion={props => <MailShared {...props} />}
  stateNavigator={stateNavigator}>
  {(style, scene, key) => (
    <div
      key={key}
      style={{opacity: style.opacity}}>
      {scene}
    </div>
  )}
</NavigationMotion>
