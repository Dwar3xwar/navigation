// Add some comments
Navigation.StateInfoConfig.build([{
    key: 'quiz', initial: 'question', states: [
        { key: 'question', route: '', 
          defaults: { count: 2 }, transitions: [
            { key: 'next', to: 'answer' }
        ]},
        { key: 'answer', route: 'thanks/{count}' }
    ]
}]);
next.onclick = function (e) {
    var data = { count: things.value };
    Navigation.StateController.navigate('next', data);
}
goBack.onclick = function (e) {
    Navigation.StateController.navigateBack(1);
}
var dialogs = Navigation.StateInfoConfig.dialogs;
var quiz = dialogs.quiz;
quiz.states.question.navigated = function () {
    question.style.display = 'block';
}
quiz.states.question.dispose = function () {
    question.style.display = 'none';
}
quiz.states.answer.navigated = function (data) {
    answer.style.display = 'block';
    count.textContent = data.count;
}
quiz.states.answer.dispose = function () {
    answer.style.display = 'none';
}
Navigation.start();
